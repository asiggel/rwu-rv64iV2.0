  \item op: opcode\_i
  \item f3: func3\_i
  \item 7: func7b5\_i
  \item z: zero\_i
  \item S: resultSrc\_o
  \item W: dMemWr\_o
  \item R: dMemRd\_o
  \item P: PCSrc\_o
  \item B: aluSrcB\_o
  \item A: aluSrcA\_o
  \item r: regWr\_o
  \item j: jump\_o
  \item im: immSrc\_o
  \item aluSel: aluSel\_o
  \item b: branch\_s\_o
  \item js: jump\_s\_o
\end{itemize}
